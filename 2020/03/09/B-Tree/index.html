B-树的操作：插入与删除

基本框架

首先，用查找算法找到关键码对应节点及其直接后继，类似于执行BST的删除操作，然后交换当前节点与直接后继，并将交换后的直接后继处的叶子节点删除。

template<typename T> bool BTree<T>::remove(T const& e){
    BTNode<T>* v=search(e);
    if(!v)  return false;
    Rank r=v->key.search(e);
    if(v->child[0]){//如果v不是叶子节点，必然存在直接后继
        BTNode<T>* u=v->child[r+1];
        while(u->child[0])  u=u->child[0];
        v->key[r]=u->key[0];
        v->key.remove(r);
        v->child.remove(r+1);
        _size--;
    }
    solveUnderFlow(v);//处理由于删除发生的下溢问题
    return true;       
}

下溢问题

三种可能：左侧足够多，右侧足够多，左右都即将下溢

template<typename T> void BTree<T>::solveUnderFlow(BTNode<T>* v){
    if((_order+1)/2<=v->child->size())//没有发生下溢，不需要变化
        return;
    BTNode<T>* p=v->parent;
    if(!p){
        if(!v->key.size()&&v->child[0]){
            _root=v->child[0];
            _root->parent=NULL;
            v->child[0]=NULL;
            release(v);
        }//树根v没有关键码，却有唯一的非空孩子
        //则可以销毁掉这个节点，并将整树高度下降一个单位       
        return;
    }
    Rank r=0;
    while(p->child[r]!=v)   r++;
    //确定v是p的第r个孩子，此时由于v可能不含关键码
    //所以不能直接通过关键码查找v
    //情况1：向左兄弟借关键码
    if(r>0){//v不是p的第一个孩子
        BTNode<T>* ls=p->child[r-1];//想想中序遍历的直接后继
        if((_order+1)/2<ls->child.size()){//左兄弟够借
            v->key.insert(0,p->key[r-1]);//先从父亲借1个
            p->key[r-1]=ls->key.remove(ls->key.size()-1);//左兄弟移给父亲
            v->child.insert(0,ls->child.remove(ls->child.size()-1));//引用节点也要相应的转移给右边的兄弟
        }
        if(v->child[0]) v->child[0]->parent=v;
        //移完了记得更新parent引用
        return;
    }
    //至此，左兄弟要么为空，要么不够借
    //情况2：向右兄弟借关键码
    if(r<p->child.size()){
        BTNode<T>* rs=p->child[r+1];
        if((_order+1)/2<rs->child.size()){
            v->key.insert(v->key.size(),p->key[r]);//从父亲借一个，插入左边最右侧
            p->key[r]=rs->key.remove(0);
            v->child.insert(v->child.size(),rs->child.remove(0));
        }
        if(v->child[v->child.size()-1])
            v->child[v->child.size()-1]->parent=v;   
    }
    //至此，右兄弟要么为空，要么不够借
    //情况3，左右兄弟要么其一为空，要么都不够借
    if(r>0){//向左兄弟合并
        BTNode<T>* ls=p->child[r-1];
        ls->key.insert(ls->key.size(),p->key.remove(r-1));
        p->child.remove(r);//将p的第r-1个关键码转入ls
        ls->child.insert(ls->child.size(),v->child.remove(0));
        if(ls->child[ls->child.size()-1])
            ls->child[ls->child.size()-1]->parent=ls;
        while(!v->key.empty()){
            ls->key.insert(ls->key.size(),v->key.remove(0));
            ls->child.insert(ls->child.size(),v->child.remove(0));
            if(ls->child[ls->child.size()-1])
                ls->child[ls->child.size()-1]->parent=ls;
        }
        release(v);
    }
    else{//向右兄弟合并
        BTNode<T>* rs=p->child[r+1];
        rs->key.insert(0,p->key.remove(r));
        p->child.remove(r);
        rs->child.insert(0,v->child.remove(v->child.size()-1));
        if(rs->child[0])
            rs->child[0]->parent=rs;
        while(!v->key.empty()){
            rs->key.insert(0,v->key.remove(v->key.size()-1));
            rs->child.insert(0,v->key.remove(v-child.size()-1));
            if(rs->child[0])
                rs->child[0]->parent=rs;
        }
        release(v);
    }
    solveUnderFlow(p);
    return;
}

复杂度：平均只有常数次的操作。极端情况为O(logn)的时间复杂度。

class MyQueue {
public:
    /** Initialize your data structure here. */
    MyQueue() {

    }
    
    /** Push element x to the back of queue. */
    void push(int x) {
        front.push(x);
    }
    
    /** Removes the element from in front of queue and returns that element. */
    int pop() {
        if(rear.empty()){
            while(!front.empty()){
                rear.push(front.top());
                front.pop();
            }
        }
        int temp=rear.top();
        rear.pop();
        return temp;
    }
    
    /** Get the front element. */
    int peek() {
        if(rear.empty()){
            while(!front.empty()){
                rear.push(front.top());
                front.pop();                
            }
        }
        return rear.top();
    }
    
    /** Returns whether the queue is empty. */
    bool empty() {
        return front.empty()&&rear.empty();
    }
private:
    stack<int> front;
    stack<int> rear;
};
